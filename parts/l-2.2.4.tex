\begin{lemma}{2.2.4}
	Lad $A \in \mat_{m,n}$. Skriv $A = [\vec{a_1}, \dotsc, \vec{a_n}]$ i 
	søjleform. 
	
	Betragt da ligningssystemet $A\vec{x} = \vec{b}$, hvor $b \in 
	\mathbb{F}^{m \times 1}$.

	Systemet er konsistent (har en løsning) $\Leftrightarrow b \in 
	\spn(\vec{a_1}, \dotsc, \vec{a_n})$.
\end{lemma}
