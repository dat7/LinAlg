\begin{definition}{2.2.11}
	Lad $V$ være et $\mathbb{F}$-vektorrum, og lad $v_1, \dots, v_n \in V$. Da
	er $v_1, \dots, v_n$ en basis for $V$ hvis:
	\begin{enumerate}
		\item $v_1, \dots, v_n$ er lineært uafhængige.
		\item $v_1, \dots, v_n$ udspænder (spanner) $V$.
	\end{enumerate}
\end{definition}
